graph = tf.Graph()
with graph.as_default():
    X = tf.placeholder(tf.float32, [None, 784])
    Y = tf.placeholder(tf.float32, [None, 10])
    phase = tf.placeholder(tf.bool)
    lr = tf.placeholder(tf.float32)
    
    w1 = tf.Variable(tf.random_normal([784, 100], stddev = 0.1))
    b1 = tf.Variable(tf.zeros([100]))
    w2 = tf.Variable(tf.random_normal([100, 10], stddev = 0.1))
    b2 = tf.Variable(tf.zeros([10]))
    
    def nn():
        h1 = tf.nn.leaky_relu(tf.add(tf.matmul(X, w1), b1))
        out = tf.nn.softmax(tf.add(tf.matmul(h1, w2), b2))
        return out, h1
    
    output = nn()
    loss = -tf.reduce_mean(tf.reduce_sum(Y*tf.log(output[0] + 1e-6) + (1-Y)*tf.log(1 - output[0] + 1e-6), 1))
    train_step = tf.train.AdamOptimizer(lr).minimize(loss)
    acc = tf.reduce_mean(tf.cast(tf.equal(tf.argmax(Y, 1), tf.argmax(output[0], 1)), tf.float32))
    
    sess = tf.Session()
    init = tf.global_variables_initializer()

graph = tf.Graph()
with graph.as_default():
    X = tf.placeholder(tf.float32, [None, 784])
    Y = tf.placeholder(tf.float32, [None, 10])
    phase = tf.placeholder(tf.bool)
    lr = tf.placeholder(tf.float32)
    
    w1 = tf.Variable(tf.random_normal([784, 100], stddev = 0.1))
    b1 = tf.Variable(tf.zeros([100]))
    w2 = tf.Variable(tf.random_normal([100, 10], stddev = 0.1))
    b2 = tf.Variable(tf.zeros([10]))
    
    def nn():
        h1_1 = tf.nn.leaky_relu(tf.add(tf.matmul(X, w1), b1))
        h1_2 = tf.layers.batch_normalization(h1_1, training = phase)
        out = tf.nn.softmax(tf.add(tf.matmul(h1_2, w2), b2))
        return out, h1_2
    
    output = nn()
    loss = -tf.reduce_mean(tf.reduce_sum(Y*tf.log(output[0] + 1e-6) + (1-Y)*tf.log(1 - output[0] + 1e-6), 1))
    update_ops = tf.get_collection(tf.GraphKeys.UPDATE_OPS)
    with tf.control_dependencies(update_ops):
        # Ensures that we execute the update_ops before performing the train_step
        train_step = tf.train.AdamOptimizer(lr).minimize(loss)
    acc = tf.reduce_mean(tf.cast(tf.equal(tf.argmax(Y, 1), tf.argmax(output[0], 1)), tf.float32))
    
    sess = tf.Session()
    init = tf.global_variables_initializer()
